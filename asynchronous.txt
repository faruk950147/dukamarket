from django.views import View
from django.utils.decorators import method_decorator
from django.views.decorators.cache import never_cache
from django.shortcuts import render
from django.contrib import messages
import logging

logger = logging.getLogger(__name__)

@method_decorator(never_cache, name='dispatch')
class HomeView(View):
    async def get(self, request):
        # ইউজার যদি লগইন করা থাকে, নাম নাও, নাহলে 'Guest'
        username = request.user.username if request.user.is_authenticated else 'Guest'
        
        # লগিং (asynchronous না হলেও চলবে)
        logger.info(f"হোম পেজে ইউজার অ্যাক্সেস করেছেন: {username}")
        
        # ম্যাসেজ দেখানো
        messages.success(request, f'স্টোরে স্বাগতম, {username}!')
        
        # template রেন্ডার করা (async compatible)
        return render(request, 'store/home.html', {})
